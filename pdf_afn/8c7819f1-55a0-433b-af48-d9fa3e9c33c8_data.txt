CREATE TABLE PC_USERS(
COD_PCUSERS INT PRIMARY KEY AUTO_INCREMENT,
LASTNAME_USER VARCHAR(100),
NAME_COMPUTER VARCHAR(20),
SERIAL_NUMBER VARCHAR(50),
CELL_NUMBER INT(11),
COD_BRAND_ID INT,
COD_MODEL_ID INT,
COD_TYPE_ID INT,
COMMENT VARCHAR(60),
DELIVER_DATE DATE,
COD_USERS_ID INT
);

CREATE TABLE BRAND(
COD_BRAND INT PRIMARY KEY AUTO_INCREMENT,
NAME VARCHAR(20)
);

ALTER TABLE PC_USERS
ADD CONSTRAINT FK_PBR
FOREIGN KEY (COD_BRAND) REFERENCES BRAND(COD_BRAND_ID);


CREATE TABLE MODEL(
COD_MODEL INT PRIMARY KEY AUTO_INCREMENT,
NAME VARCHAR(20)
);

ALTER TABLE PC_USERS
ADD CONSTRAINT FK_PMO
FOREIGN KEY (COD_MODEL_ID) REFERENCES MODEL(COD_MODEL);

CREATE TABLE TYPE(
COD_TYPE INT PRIMARY KEY AUTO_INCREMENT,
NAME VARCHAR(20)
);

ALTER TABLE PC_USERS
ADD CONSTRAINT FK_PTY
FOREIGN KEY (COD_TYPE_ID) REFERENCES TYPE(COD_TYPE);

CREATE TABLE USERS(
COD_USERS INT PRIMARY KEY AUTO_INCREMENT,
NICK_NAME VARCHAR(30),
U_PASSWORD VARCHAR(80),
);

ALTER TABLE PC_USERS
FOREIGN KEY (COD_USERS_ID) REFERENCES USERS(COD_USERS);

CREATE TABLE USERS_PROFILES(
COD_PROFILE INT PRIMARY KEY AUTO_INCREMENT,
COD_USERS_ID INT,
COD_PROFILE_ID INT
);

CREATE TABLE PROFILES(
COD_PROFILE INT PRIMARY KEY AUTO_INCREMENT,
NAME VARCHAR(30)
);

ALTER TABLE USERS_PROFILES
FOREIGN KEY (COD_PROFILE_ID) REFERENCES PROFILES(COD_PROFILE);

ALTER TABLE USERS_PROFILE
FOREIGN KEY (COD_USERS_ID) REFERENCES USERS(COD_USERS);
